import java.io.*;
import java.util.*;
import javax.servlet.*;
import javax.servlet.http.*;

public class Hello extends HttpServlet {

    public void doGet(HttpServletRequest request, HttpServletResponse response)
    throws IOException, ServletException
    {
        response.setContentType("text/html");
        PrintWriter out = response.getWriter();
        out.println("<html>");
        out.println("<head>");
        out.println("<title>Request Hello</title>");
        out.println("</head>");
        out.println("<body>");
        out.println("<h3>Hello</h3>");
        out.println("Parameters in this request:<br>");
        if (userName != null || password != null) {
            out.println("User Name:");
            out.println(" = " + HTMLFilter.filter(userName) + "<br>");
            out.println("Password:");
            out.println(" = " + HTMLFilter.filter(password));
            out.println("Keyname:");
            out.println(" = " + HTMLFilter.filter(keyname));
        } else {
            out.println("No Parameters, Please enter some");
        }
        out.println("<P>");
        out.print("<form action=\"");
        out.print("Hello\" ");
        out.println("method=POST>");
        out.println("User Name:");
        out.println("<input type=text size=20 name=username>");
        out.println("<br>");
        out.println("Password:");
        out.println("<input type=text size=20 name=password>");
        out.println("<br>");
        out.println("Key Name:");
        out.println("<input type=text size=20 name=keyname>");
        out.println("<br>");
        out.println("<input type=submit>");
        out.println("</form>");
        out.println("</body>");
        out.println("</html>");
    }

    public void doPost(HttpServletRequest request, HttpServletResponse res)
    throws IOException, ServletException
    {
        doGet(request, response);
    }
}
